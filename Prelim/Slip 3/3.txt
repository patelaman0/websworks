// Question: Write a program in C# to create a function to swap the values of two integers.

using System;

class Program
{
    static void Main()
    {
        int a = 5, b = 10;

        Console.WriteLine($"Before swap: a = {a}, b = {b}");
        
        Swap(ref a, ref b);

        Console.WriteLine($"After swap: a = {a}, b = {b}");
    }

    static void Swap(ref int x, ref int y)
    {
        int temp = x;
        x = y;
        y = temp;
    }
}
