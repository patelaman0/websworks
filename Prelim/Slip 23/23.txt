// Question: Write a C# (Win Form) program to check whether the entered string is a palindrome or not.

using System;
using System.Windows.Forms;

namespace PalindromeCheck
{
    public partial class Form1 : Form
    {
        public Form1()
        {
            InitializeComponent();
        }

        private void btnCheck_Click(object sender, EventArgs e)
        {
            string input = txtInput.Text;
            string reversed = string.Join("", input.ToCharArray().Reverse());

            if (input.Equals(reversed, StringComparison.OrdinalIgnoreCase))
            {
                lblResult.Text = "The string is a palindrome.";
            }
            else
            {
                lblResult.Text = "The string is not a palindrome.";
            }
        }
    }
}
