// Question: Write a C# (Win Form) .NET program to accept sentences in a text box and count the number of words 
// and display the count in a message box.

using System;
using System.Windows.Forms;

namespace WordCounterApp
{
    public partial class Form1 : Form
    {
        public Form1()
        {
            InitializeComponent();
        }

        private void btnCountWords_Click(object sender, EventArgs e)
        {
            string sentence = txtSentence.Text;
            if (!string.IsNullOrWhiteSpace(sentence))
            {
                int wordCount = sentence.Split(new[] { ' ', '\t', '\n' }, StringSplitOptions.RemoveEmptyEntries).Length;
                MessageBox.Show($"The number of words: {wordCount}");
            }
            else
            {
                MessageBox.Show("Please enter a sentence.");
            }
        }
    }
}
