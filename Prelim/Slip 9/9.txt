// Question: Write a C# (Win Form).NET program to design the following form with menus.
// Color (Red, Blue, Green), Window (Maximize, Minimize, Restore) and Exit. 
// The form control should be affected by menu selection (e.g., change color, maximize, etc.).

using System;
using System.Windows.Forms;

namespace MenuFormApp
{
    public partial class Form1 : Form
    {
        public Form1()
        {
            InitializeComponent();
            InitializeMenus();
        }

        private void InitializeMenus()
        {
            // Create Menus
            MenuStrip menuStrip = new MenuStrip();
            ToolStripMenuItem colorMenu = new ToolStripMenuItem("Color");
            ToolStripMenuItem windowMenu = new ToolStripMenuItem("Window");
            ToolStripMenuItem exitMenu = new ToolStripMenuItem("Exit");

            // Add menu items for color
            colorMenu.DropDownItems.Add("Red", null, (sender, e) => this.BackColor = System.Drawing.Color.Red);
            colorMenu.DropDownItems.Add("Blue", null, (sender, e) => this.BackColor = System.Drawing.Color.Blue);
            colorMenu.DropDownItems.Add("Green", null, (sender, e) => this.BackColor = System.Drawing.Color.Green);

            // Add menu items for window
            windowMenu.DropDownItems.Add("Maximize", null, (sender, e) => this.WindowState = FormWindowState.Maximized);
            windowMenu.DropDownItems.Add("Minimize", null, (sender, e) => this.WindowState = FormWindowState.Minimized);
            windowMenu.DropDownItems.Add("Restore", null, (sender, e) => this.WindowState = FormWindowState.Normal);

            // Add Exit functionality
            exitMenu.Click += (sender, e) => Application.Exit();

            // Add menus to MenuStrip
            menuStrip.Items.Add(colorMenu);
            menuStrip.Items.Add(windowMenu);
            menuStrip.Items.Add(exitMenu);
            Controls.Add(menuStrip);
            MainMenuStrip = menuStrip;
        }
    }
}
