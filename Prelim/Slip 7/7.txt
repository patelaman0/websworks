/*
 * Slip 2A: Write a C# .Net program to move the Text “Pune University” continuously from Left
 * to Right and Vice Versa.
 */
using System;
using System.Windows.Forms;

namespace Slip2A
{
    public partial class Form1 : Form
    {
        private int direction = 1; // 1 for right, -1 for left

        public Form1()
        {
            InitializeComponent();
            timer1.Interval = 50;
            timer1.Start();
        }

        private void timer1_Tick(object sender, EventArgs e)
        {
            if (lblText.Left + direction > ClientSize.Width - lblText.Width || lblText.Left + direction < 0)
                direction *= -1;

            lblText.Left += direction;
        }
    }
}
