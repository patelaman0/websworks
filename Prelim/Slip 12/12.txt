// Question: Write a program in C#.Net to find the sum of all elements of the array.

using System;

class Program
{
    static void Main()
    {
        int[] numbers = { 1, 2, 3, 4, 5 };
        int sum = 0;

        foreach (int number in numbers)
        {
            sum += number;
        }

        Console.WriteLine("Sum of all elements in the array: " + sum);
    }
}
