// Question: Write a program in C# to create a function to display the n terms of Fibonacci Sequence.

using System;

class Program
{
    static void Main()
    {
        Console.Write("Enter the number of terms for Fibonacci Sequence: ");
        int n = Convert.ToInt32(Console.ReadLine());

        DisplayFibonacci(n);
    }

    static void DisplayFibonacci(int n)
    {
        int a = 0, b = 1, next;

        Console.WriteLine("Fibonacci Sequence:");
        for (int i = 0; i < n; i++)
        {
            if (i <= 1)
                next = i;
            else
            {
                next = a + b;
                a = b;
                b = next;
            }

            Console.Write(next + " ");
        }
    }
}
