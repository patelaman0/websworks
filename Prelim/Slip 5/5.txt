/*
 * Slip 2B: Write a C#.Net program to create a base class Department and derived classes Sales and
 * Human Resource. Accept the details of both departments and display them in proper format.
 */
using System;

namespace Slip2B
{
    public class Department
    {
        public string Name { get; set; }
        public int Employees { get; set; }

        public virtual void Display()
        {
            Console.WriteLine($"Department: {Name}, Employees: {Employees}");
        }
    }

    public class Sales : Department
    {
        public double Revenue { get; set; }

        public override void Display()
        {
            base.Display();
            Console.WriteLine($"Revenue: {Revenue}");
        }
    }

    public class HumanResource : Department
    {
        public int OpenPositions { get; set; }

        public override void Display()
        {
            base.Display();
            Console.WriteLine($"Open Positions: {OpenPositions}");
        }
    }

    class Program
    {
        static void Main(string[] args)
        {
            Sales sales = new Sales { Name = "Sales", Employees = 10, Revenue = 50000 };
            HumanResource hr = new HumanResource { Name = "HR", Employees = 5, OpenPositions = 3 };

            sales.Display();
            hr.Display();
        }
    }
}
