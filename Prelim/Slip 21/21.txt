/*
 * Slip 4A: Design a C#.Net form to pick a date from DateTimePicker Control and display day, month
 * and year in separate text boxes.
 */
using System;
using System.Windows.Forms;

namespace Slip4A
{
    public partial class Form1 : Form
    {
        public Form1()
        {
            InitializeComponent();
        }

        private void dateTimePicker1_ValueChanged(object sender, EventArgs e)
        {
            txtDay.Text = dateTimePicker1.Value.Day.ToString();
            txtMonth.Text = dateTimePicker1.Value.Month.ToString();
            txtYear.Text = dateTimePicker1.Value.Year.ToString();
        }
    }
}
