/*
 * Slip 1A: Write a C# .Net Program to display the numbers continuously in Text Box by
 * clicking on Button.
 */
using System;
using System.Windows.Forms;

namespace Slip1A
{
    public partial class Form1 : Form
    {
        private int counter = 0;

        public Form1()
        {
            InitializeComponent();
        }

        private void btnDisplay_Click(object sender, EventArgs e)
        {
            txtNumbers.Text += counter.ToString() + " ";
            counter++;
        }
    }
}
